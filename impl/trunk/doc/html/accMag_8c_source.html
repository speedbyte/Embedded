179 /*!**********************************************************************
180  * \author Juergen Schmidt (juscgs00)
181  * \date 2014/05/08
182  *
183  * \brief Wrapper function to write byte stream to registers of LSM303D
184  * \details This function is a wrapper to the I2C Interface to write a
185  * byte stream to a series of registers of the Accelerometer/
186  * Magnetometer (LSM303D). The first register address can be
187  * defined in the parameter list of the function. After writing
188  * the first byte to the start address, the register address
189  * will be incremented.
190  *
191  * \param[in] f_startRegisterAddr_ui8 specifies the first register
192  * address to return
193  * \param[in] f_byteStreamToWrite_pui8 points to an array of unsigned
194  * 8bit data elements (input byte stream buffer)
195  * \param[out] returns a boolean value, indicating the occurrence of
196  * failures\n
197  * 0 .... indicates success (no errors)\n
198  * 1 .... indicates failed (some errors)
199  *
200  * \internal
201  * CHANGELOG:
202  * 2015/05/13 juscgs00:
203  * - changed malloc/free approach to fixed size array (with define to set
204  * size of array in header)
205  * \endinternal
206  ***********************************************************************/
207 static inline unsigned int l_writeI2cByteStream_bl( unsigned char f_startRegisterAddr_ui8, const unsigned char* const f_byteStreamToWrite_pui8, unsigned int f_numOfBytes_ui32 )
208 {
209  unsigned int l_writeSucceeded_bl = M_HAL_ACCMAG_SUCCESS_BL; // success state as default
210  unsigned char l_bufCtr_ui8 = 0; // loop counter to assemble payload and register address
211  unsigned char l_byteStreamBuffer_rgXui8[M_HAL_ACCMAG_WRITE_BYTESTREAM_BUFFER_SIZE_UI8];
212  /*
213  * - ensure at least one data byte to write
214  * - prevent a buffer overflow of l_byteStreamBuffer_rgXui8
215  * --> payload data bytes + register address (subaddress)
216  */
217  if ( (f_numOfBytes_ui32 < 1)
218  || (f_numOfBytes_ui32 > M_HAL_ACCMAG_WRITE_BYTESTREAM_BUFFER_SIZE_UI8-1) )
219  {
220  return M_HAL_ACCMAG_FAILED_BL;
221  }
222 
223  /*
224  * prepare data to write (register + data)
225  * (1) ensure byte stream writing (MSB of register address shall be 1)
226  * -> Mask MSB to one!
227  * -> LSM303D will auto-increment the register address for each byte
228  * of data
229  * (2) copy data to write to local buffer, start at second byte position
230  * (first byte position is the register address)
231  */
232  l_byteStreamBuffer_rgXui8[0] = f_startRegisterAddr_ui8 | 0b10000000;
233  for(l_bufCtr_ui8=0; l_bufCtr_ui8 < f_numOfBytes_ui32; l_bufCtr_ui8++)
234  {
235  l_byteStreamBuffer_rgXui8[l_bufCtr_ui8+1] = f_byteStreamToWrite_pui8[l_bufCtr_ui8];
236  }
237 
238  //write register to I2C
239  if ( g_lldI2c_WriteI2c_bl(M_HAL_ACCMAG_I2CADDR_UI8, l_byteStreamBuffer_rgXui8, f_numOfBytes_ui32 + 1) != M_HAL_ACCMAG_SUCCESS_BL )
240  {
241  // error in I2C access
242  l_writeSucceeded_bl = M_HAL_ACCMAG_FAILED_BL;
243  }
244 
245  return l_writeSucceeded_bl;
246 }

250  * \date 2014/05/10
251  *
252  * \brief Complement of twos of a 16 bit integer
253  * \details This function computes the two's complement of two 8bit
254  * integers (most significant and least significant bytes)
255  *
256  * \param[in] f_lsb_ui8 is the least significant byte
257  * \param[in] f_msb_ui8 is the most significant byte
258  * \param[out] returns a unsigned 16bit integer of the two's complement
259  *
260  * \internal
261  * CHANGELOG:
262  * none
263  * \endinternal
264  ***********************************************************************/
265 static inline signed short l_twoCompOf16Bit_ui16( unsigned char f_lsb_ui8, unsigned char f_msb_ui8 )
266 {
267  signed short l_compOfTwo_ui16 = 0; //init variable with zeros
268 
269  /*
270  * 1. assemble 16bit value of two 8bit chunks
271  * 2. invert 16bit
272  * 3. add 1 to 16bit value
273  * 4. mask only 16bit range (to avoid overflows)
274  * --> conversion of two's complement
275  */
276  l_compOfTwo_ui16 = (signed short)( ( ~((int)(f_msb_ui8 << 8) + (int)(f_lsb_ui8)) + (int)1 ) & (int)0xFFFF );
277 
278  return l_compOfTwo_ui16;

285  * \brief Converts raw sensor values of the magnetometer into SI unit
286  * \details This function converts the raw sensor values (signed 16bit
287  * integers)of the magnetometer into SI unit (64bit float). The
288  * unit for the magnetometer is Tesla\n
289  * (SI representation: Tesla = \f[ \frac{kg}{s^2\dotA} \f])
290  *
291  * \param[in] f_rawMagVal_i16 is raw signed 16bit sensor value
292  * \param[in] f_scalingValue_ui8 is the absolute value of the scale
293  * defined maximal sensor value range (e.g. "+/-4 Gauss" will
294  * result in "4")
295  * \param[out] returns a 64bit float value in SI units\n
296  * (Tesla = \f[ \frac{kg}{s^2\dotA} \f])
297  *
298  * \internal
299  * CHANGELOG:
300  * none
301  * \endinternal
302  ***********************************************************************/
303 static inline double l_convertMagRawToSIUnit_f64( signed short f_rawMagVal_i16, unsigned char f_scalingValueGauss_ui8 )
304 {
305  double l_outputValue_f64 = 0;
306  const double l_gaussToSI_f64 = 0.1E-3; //1Gauss = 0.1mT = 0.1*10^(-3)T
307 
308  // unitless [-1,1]
309  l_outputValue_f64 = (double)f_rawMagVal_i16 / (double)SHRT_MAX;
310 
311  // Units: 1 * T*Gauss^(-1) * Gauss = m*s^(-2)
312  l_outputValue_f64 = l_outputValue_f64 * l_gaussToSI_f64 * (double)f_scalingValueGauss_ui8;
313 
314  return l_outputValue_f64; //Tesla
315 }

317 /*!**********************************************************************
318  * \author Juergen Schmidt (juscgs00)
319  * \date 2014/05/11
320  *
321  * \brief Converts raw sensor values of the accelerometer into SI unit
322  * \details This function converts the raw sensor values (signed 16bit
323  * integers)of the accelerometer into SI unit (64bit float). The
324  * unit for the accelerometer is \f[ \frac{m}{s^2} \f]
325  *
326  * \param[in] f_rawMagVal_i16 is raw signed 16bit sensor value
327  * \param[in] f_scalingValue_ui8 is the absolute value of the max. scale
328  * defined in the sensor's control registers.\n
329  * Example: maximal sensor range "+/- 8G" will result in "8"
330  * \param[out] returns a 64bit float value in SI units (\f[ \frac{m}{s^2} \f])
331  *
332  * \internal
333  * CHANGELOG:
334  * none
335  * \endinternal
336  ***********************************************************************/
337 static inline double l_convertAccRawToSIUnit_f64( signed short f_rawAccVal_i16, unsigned char f_scalingValueG_ui8 )
338 {
339  double l_outputValue_f64 = 0;
340  const double l_gravityConst_f64 = 9.81; // m*(s^-2) per G
341 
342  // unitless [-1,1]
343  l_outputValue_f64 = (double)f_scalingValueG_ui8 * (double)f_rawAccVal_i16 / (double)SHRT_MAX;
344 
345  // Units: 1 * m*s^(-2)*G^(-1) * G = m*s^(-2)
346  l_outputValue_f64 = l_outputValue_f64 * l_gravityConst_f64;
347 
348  return l_outputValue_f64; // m*(s^-2)

452 /*!**********************************************************************
453  * \author Juergen Schmidt (juscgs00)
454  * \date 2014/05/05
455  *
456  * \brief Initialize sensor
457  * \details Sends all commands on I2C to initialize the acc.- and
458  * magnetometer. It uses on big 8 byte config buffer to set all
459  * necessary bits. This way, a burst write to the control
460  * registers of LSM303D is performed to reduce the I2C bus load.
461  *
462  * \param[out] returns a boolean value, indicating the occurrence of
463  * failures\n
464  * 0 .... indicates success (no errors)\n
465  * 1 .... indicates failed (some errors)
466  *
467  * \internal
468  * CHANGELOG:
469  * none
470  * \endinternal
471  ***********************************************************************/
472 unsigned int g_halAccmag_initSensor_bl(void)
473 {
474  unsigned char l_configRegisters_rg8ui8[8]; //buffer for 8 control register bytes
475 
476  //Identify sensor
477  if ( l_readI2cByte_ui8( M_HAL_ACCMAG_WHO_AM_I_UI8 ) != M_HAL_ACCMAG_WHO_AM_I_PATTERN_UI8 )
478  {
479  // sensor could not be identified --> abort initialization
480  return M_HAL_ACCMAG_FAILED_BL;
481  }
482 
483  /*
484  * ACCELEROMETER
485  */
486  // CTRL0: no FIFO (default values)
487  l_configRegisters_rg8ui8[0] = 0b00000000;
488 
489  // CTRL1: Enable accelerometer, 800Hz sampling
490  l_configRegisters_rg8ui8[1] = ( M_HAL_ACCMAG_CTRL1_MASK_AXEN_UI8
491  | M_HAL_ACCMAG_CTRL1_MASK_AYEN_UI8
492  | M_HAL_ACCMAG_CTRL1_MASK_AZEN_UI8
493  | M_HAL_ACCMAG_CTRL1_MASK_ACC_RATE_800HZ_UI8);
494 
495  // CTRL2 register (default values): Set scale +/-8G (accelerometer) & Anti-Alias-Filter ~200Hz
496  l_configRegisters_rg8ui8[2] = ( M_HAL_ACCMAG_CTRL2_MASK_ACC_SCALE_8G_UI8
497  | M_HAL_ACCMAG_CTRL2_MASK_ACC_BANDWIDTH_194HZ_UI8 );
498 
499  // CTRL3 register (default values): Disable all interrupts
500  l_configRegisters_rg8ui8[3] = 0b00000000;
501 
502  //CTRL4 register (default values)
503  l_configRegisters_rg8ui8[4] = 0b00000000;
504 
505  /*
506  * MAGNETOMETER
507  */
508  // CTRL5 register: Disable Temp.-sensor, set high-resolution, set 100Hz data rate
509  l_configRegisters_rg8ui8[5] = ( M_HAL_ACCMAG_CTRL5_MASK_MAG_RES_HIGH_UI8
510  | M_HAL_ACCMAG_CTRL5_MASK_MAG_RATE_100HZ_UI8 );
511 
512  // CTRL6 register: set scale +/-0.4mT
513  l_configRegisters_rg8ui8[6] = M_HAL_ACCMAG_CTRL6_MASK_MAG_SCALE_4GAUSS_UI8;
514 
515  // CTRL7 register (default values): set continuous-conversion mode (magnetometer)
516  l_configRegisters_rg8ui8[7] = M_HAL_ACCMAG_CTRL7_MASK_MAG_SENSMODE_CONTINOUS_UI8;
517 
518 
519  // send config
520  if ( l_writeI2cByteStream_bl(M_HAL_ACCMAG_CTRL0_UI8, l_configRegisters_rg8ui8, sizeof(l_configRegisters_rg8ui8) ) == M_HAL_ACCMAG_FAILED_BL )
521  {
522  // setting register CTRL1 failed
523  return M_HAL_ACCMAG_FAILED_BL;
524  }
525 
526  return M_HAL_ACCMAG_SUCCESS_BL;
527 }
